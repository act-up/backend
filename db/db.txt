CREATE TABLE active_issues (
    issue_id SERIAL PRIMARY KEY,
    issue_name VARCHAR (200) NOT NULL,
    issue_target VARCHAR (100) NOT NULL,
    issue_description VARCHAR (2500) NOT NULL,
    contact_url VARCHAR (100),
    twitter_username VARCHAR (50),
    contact_email VARCHAR (50),
    form_letter VARCHAR (3000) NOT NULL,
    twitter_dm_form VARCHAR (10000)
);

CREATE TABLE archived_issues (
) INHERITS (active_issues);


CREATE TABLE results (
    issue_id SERIAL PRIMARY KEY,
    num_forms_sent INT,
    num_emails_sent INT,
    num_twitter_dms_sent INT,
    resolution VARCHAR (10000)
);


CREATE TABLE suggested (
    contact_name VARCHAR (75) NOT NULL,
    contact_email VARCHAR (50) NOT NULL,
    issue_target VARCHAR (100) NOT NULL,
    issue_summary VARCHAR (1000) NOT NULL,
    issue_description VARCHAR (10000) NOT NULL,
    other_comments VARCHAR (10000)
);
